/* mytheme/functions.php */
add_filter( 'acf/load_field/key=field_53dcf25246f92', function($field) {
	$posts = get_posts('post_type=post');
	foreach($posts as $post) {
		$field['choices'][$post->post_name] = $post->post_title;
	}
	return $field;
});
